      \begin{itemize}
        \item Abrasion
          \begin{itemize}
            \item Debris entrained by the ice scratch the bedrock
          \end{itemize}
        \bigskip
        \item Plucking
          \begin{itemize}
            \item Fluctuations of water pressure induce fracturing
            \item Loose blocks are carried away by the ice
